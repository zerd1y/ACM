P1048 [NOIP2005 普及组] 采药
```cpp
#include<bits/stdc++.h>
using namespace std;
int vol[101];
int value[101];
int dp[101][1001];
int main()
{
	int m,t;
	cin>>t>>m;
	for (int i=1;i<=m;i++){
		cin>>vol[i]>>value[i];
	}
	for (int i=1;i<=m;i++){
		for (int j=0;j<=t;j++){
			if(vol[i]>j){
				dp[i][j]=dp[i-1][j];
			}else {
				dp[i][j]=max(dp[i-1][j],dp[i-1][j-vol[i]]+value[i]);
			}
		}
	}
	cout<<dp[m][t];
	return 0;
}
```
